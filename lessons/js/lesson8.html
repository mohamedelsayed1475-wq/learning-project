💾 التخزين المحلي وبناء تطبيق قائمة المهام

الدرس الختامي! تعلم كيف تحتفظ ببيانات المستخدم وتمنعها من الاختفاء عند تحديث الصفحة باستخدام التخزين المحلي (localStorage)، وسنجمع كل ما تعلمناه لبناء تطبيق متكامل لقائمة المهام.

💾 ما هو التخزين المحلي (localStorage)؟

هو عبارة عن مساحة تخزين صغيرة يوفرها متصفح الويب للموقع، تتيح للمطورين تخزين البيانات على شكل مفتاح وقيمة (Key - Value Pairs) بشكل مستمر.

💡 على عكس ملفات تعريف الارتباط (Cookies)، فإن بيانات localStorage تظل محفوظة للأبد حتى لو أغلق المستخدم المتصفح أو أطفأ الحاسوب، ولا تحذف إلا إذا قام المستخدم بمسح بيانات التصفح بنفسه.

// 1️⃣ حفظ البيانات (تخزين نص)
localStorage.setItem('username', 'أحمد');

// 2️⃣ استرجاع البيانات المحفوظة
const storedName = localStorage.getItem('username');
console.log(storedName); // أحمد

// 3️⃣ حذف مفتاح معين
localStorage.removeItem('username');

// 4️⃣ مسح كل بيانات التخزين المحلي للموقع
localStorage.clear();

🔄 حفظ واستعادة البيانات المعقدة (JSON)

التخزين المحلي يقبل تخزين نصوص فقط (Strings). إذا حاولت حفظ مصفوفة أو كائن، فسيقوم بتحويله لنص غير صالح "[object Object]". لحل هذه المشكلة نستخدم كائن JSON. 

const userProfile = {
  name: 'سارة',
  score: 100
};

// ❌ طريقة خاطئة:
localStorage.setItem('user', userProfile);
console.log(localStorage.getItem('user')); // "[object Object]"

// ✅ الطريقة الصحيحة:
// 1. تحويل الكائن إلى نص JSON (Stringify)
const stringifiedData = JSON.stringify(userProfile);
localStorage.setItem('user', stringifiedData);

// 2. استرجاع النص وتحويله مجدداً لكائن (Parse)
const retrievedText = localStorage.getItem('user');
const parsedUser = JSON.parse(retrievedText);
console.log(parsedUser.name); // سارة

🛠️ تطبيق عملي: قائمة المهام (Todo List Project)

الآن، سنبني معاً قائمة مهام تفاعلية تتيح للمستخدم:

  • ➕ إضافة مهام جديدة.
  • ✔️ تحديد المهمة كمنجزة أو غير منجزة بنقرة واحدة.
  • ❌ حذف المهمة نهائياً.
  • 💾 حفظ التغييرات في التخزين المحلي حتى لا تضيع عند التحديث.

💻 الكود الكامل للتطبيق:

// 1. تحديد العناصر من الـ DOM
const input = document.getElementById('new-todo-input');
const addButton = document.getElementById('add-todo-btn');
const todoList = document.getElementById('todos-container');

// 2. مصفوفة لتخزين المهام، مع محاولة تحميلها من localStorage أولاً
let tasks = JSON.parse(localStorage.getItem('myTasks')) || [];

// 3. دالة لحفظ المصفوفة الحالية في localStorage
function saveTasks() {
  localStorage.setItem('myTasks', JSON.stringify(tasks));
}

// 4. دالة عرض وتحديث القائمة في الـ DOM
function renderTasks() {
  todoList.innerHTML = ''; // تفريغ الحاوية أولاً

  if (tasks.length === 0) {
    todoList.innerHTML = '<li class="todo-empty-msg">لا توجد مهام حالياً. أضف مهمة جديدة!</li>';
    return;
  }

  tasks.forEach((task, index) => {
    // إنشاء عنصر قائمة <li>
    const li = document.createElement('li');
    li.className = `todo-item ${task.completed ? 'completed' : ''}`;

    // بناء محتوى العنصر
    li.innerHTML = `
      ${task.text}
      
    `;

    todoList.appendChild(li);
  });
}

// 5. دالة إضافة مهمة جديدة
function addTask() {
  const text = input.value.trim(); // إزالة الفراغات الزائدة
  if (text === '') return;          // تجاهل لو الحقل فارغ

  tasks.push({
    text: text,
    completed: false
  });

  input.value = ''; // تفريغ الحقل
  saveTasks();      // الحفظ
  renderTasks();    // إعادة الرسم
}

// 6. دالة تغيير حالة الانجاز
function toggleTask(index) {
  tasks[index].completed = !tasks[index].completed;
  saveTasks();
  renderTasks();
}

// 7. دالة حذف مهمة
function deleteTask(index) {
  tasks.splice(index, 1); // حذف عنصر واحد من الفهرس المحدد
  saveTasks();
  renderTasks();
}

// 8. ربط حدث النقر والضغط على زر Enter للإضافة
addButton.addEventListener('click', addTask);
input.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') addTask();
});

// 9. تشغيل العرض الأولي عند تحميل الصفحة
renderTasks();

🔀 الشروط والتكرار

تعلم كيف تجعل كودك يتخذ قرارات بـ if/else، وكيف تكرر تنفيذ كود بشكل متكرر بحلقات for وwhile!

🔀 جمل الشرط (Conditionals)

// if - if/else - if/else if/else
const score = 75;

// جملة if بسيطة
if (score >= 60) {
  console.log('ناجح! 🎉');
}
/*
  if = إذا
  (score >= 60) = الشرط: هل النتيجة 60 أو أكثر؟
  {} = تنفذ هذا الكود لو الشرط صحيح
*/

// if/else - إذا/وإلا
if (score >= 60) {
  console.log('ناجح! 🎉');
} else {
  console.log('راسب! 😢');
}

// if/else if/else - شروط متعددة
if (score >= 90) {
  console.log('ممتاز! 🌟');
} else if (score >= 75) {
  console.log('جيد جداً! ✅');
} else if (score >= 60) {
  console.log('جيد 👍');
} else {
  console.log('يحتاج مراجعة 📚');
}

عمليات المقارنة

// عمليات المقارنة
console.log(5 > 3);   // true - أكبر من
console.log(5 < 3);   // false - أصغر من
console.log(5 >= 5);  // true - أكبر أو يساوي
console.log(5 <= 4);  // false - أصغر أو يساوي
console.log(5 === 5); // true - يساوي تماماً (قيمة ونوع)
console.log(5 !== 3); // true - لا يساوي
console.log(5 == '5');// true - يساوي (بدون تحقق من النوع)

// العمليات المنطقية
const age = 20;
const hasID = true;

// AND (&&) = كلا الشرطان صحيحان
if (age >= 18 && hasID) {
  console.log('يمكن الدخول ✅');
}

// OR (||) = أحد الشرطين صحيح
if (age < 18 || !hasID) {
  console.log('لا يمكن الدخول ❌');
}

// NOT (!) = عكس الشرط
if (!hasID) {
  console.log('لا يوجد هوية!');
}

🔄 حلقات التكرار (Loops)

// for loop - للعدد المعروف مسبقاً
for (let i = 1; i <= 5; i++) {
  console.log(`رقم ${i}`);
}
/*
  let i = 1  → قيمة البداية
  i <= 5     → الشرط: استمر طالما i أصغر أو يساوي 5
  i++        → زد i بمقدار 1 في كل دورة
  
  النتيجة: رقم 1، رقم 2، رقم 3، رقم 4، رقم 5
*/

// while - يستمر طالما الشرط صحيح
let count = 0;
while (count < 3) {
  console.log(`المحاولة ${count + 1}`);
  count++;  // مهم جداً! وإلا حلقة لانهائية
}

// do...while - ينفذ مرة واحدة على الأقل
let x = 10;
do {
  console.log('تنفّذت!');
  x++;
} while (x < 5);
// سيطبع 'تنفّذت!' مرة واحدة رغم أن x=10 > 5

// for...of - للمصفوفات والقوائم
const fruits = ['تفاح', 'موز', 'برتقال'];
for (const fruit of fruits) {
  console.log(fruit);
}
// تفاح، موز، برتقال

// break - إيقاف الحلقة فوراً
for (let i = 0; i < 10; i++) {
  if (i === 5) break; // توقف عند 5
  console.log(i);
}
// يطبع: 0, 1, 2, 3, 4

// continue - تخطي التكرار الحالي
for (let i = 0; i < 5; i++) {
  if (i === 2) continue; // تخطي 2
  console.log(i);
}
// يطبع: 0, 1, 3, 4

⚙️ الدوال والنطاق

الدوال هي بلوكات من الكود يمكن إعادة استخدامها. تعلم كيف تكتب دوالاً فعّالة وتفهم مفهوم النطاق (Scope)!

⚙️ ما هي الدالة؟

💡 الدالة = وصفة طبخ قابلة للإعادة: تعرّفها مرة وتستخدمها كثيراً!

بدل كتابة نفس الكود مرات ومرات، ضعه في دالة واستدعها كلما احتجته.

// 1. الدالة التقليدية (Function Declaration)
function greet(name) {
  console.log(`مرحباً ${name}! 👋`);
}
/*
  function = الكلمة الأساسية
  greet = اسم الدالة
  (name) = المعامل (Parameter)
  {} = جسم الدالة
*/

// استدعاء الدالة
greet('أحمد');   // مرحباً أحمد! 👋
greet('سارة');   // مرحباً سارة! 👋
greet('محمد');   // مرحباً محمد! 👋
// نفس الدالة، نتائج مختلفة!

// 2. دالة تُعيد قيمة (return)
function add(a, b) {
  return a + b;  // ترجع الناتج
}

const result = add(5, 3);
console.log(result);  // 8
/*
  return = إرجع هذه القيمة للمستدعي
  الدالة تتوقف عند return
*/

🏹 الدوال الحديثة

// Function Expression (دالة كقيمة)
const multiply = function(a, b) {
  return a * b;
};
console.log(multiply(4, 5)); // 20

// Arrow Function (الدالة السهمية) ← الأكثر شيوعاً الآن
const square = (x) => {
  return x * x;
};

// مختصرة أكثر لو سطر واحد
const square2 = x => x * x;
console.log(square2(5));  // 25

// لو معاملان يلزم قوسان
const divide = (a, b) => a / b;
console.log(divide(10, 2)); // 5

// القيم الافتراضية للمعاملات
function greet(name = 'زائر', emoji = '👋') {
  return `مرحباً ${name} ${emoji}`;
}
console.log(greet());          // مرحباً زائر 👋
console.log(greet('أحمد'));    // مرحباً أحمد 👋
console.log(greet('سارة', '🌟')); // مرحباً سارة 🌟

🌐 النطاق (Scope)

// النطاق العام (Global Scope)
let globalVar = 'أنا عالمي';

function testScope() {
  // النطاق المحلي (Local Scope)
  let localVar = 'أنا محلي';
  console.log(globalVar);  // ✅ يمكن الوصول للعالمي
  console.log(localVar);   // ✅ يمكن الوصول للمحلي
}

testScope();
console.log(globalVar);  // ✅ يعمل
// console.log(localVar);  ❌ خطأ! localVar خارج النطاق

/*
  القاعدة: المتغيرات الداخلية تُرى من الخارج
  لكن الخارجية تُرى من الداخل (وراثة النطاق)
*/

🎓 ما تعلمناه في هذا الدرس

  • ✅ الدالة التقليدية بـ function وreturn
  • ✅ Function Expression وArrow Function
  • ✅ المعاملات (Parameters) والقيم الافتراضية
  • ✅ النطاق: Global مقابل Local Scope
  • ✅ الدوال كمكتبة كود قابل للإعادة

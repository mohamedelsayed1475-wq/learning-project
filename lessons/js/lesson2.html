📦 المتغيرات وأنواع البيانات

المتغيرات هي أوعية لحفظ البيانات في الذاكرة. تعلم كيف تُعرِّف المتغيرات بـ let وconst وvar، وتفهم أنواع البيانات الأساسية في JavaScript!

📦 ما هو المتغير؟

💡 المتغير = صندوق له اسم يحفظ فيه قيمة ما

تخيل أن لديك صناديق: صندوق اسمه "userName" يحفظ "أحمد"، وصندوق اسمه "age" يحفظ 25

// تعريف متغير بـ let (الأكثر استخداماً)
let userName = 'أحمد';
/*
  let = الكلمة الأساسية لتعريف متغير
  userName = اسم المتغير (اخترناه نحن)
  = تعني التعيين (assignment)
  'أحمد' = القيمة المحفوظة
*/

// يمكن تغيير قيمة let لاحقاً
userName = 'محمد';  // ✅ مسموح
console.log(userName);  // 'محمد'

// تعريف ثابت بـ const
const PI = 3.14159;
/*
  const = constant = ثابت لا يتغير
  لا يمكن إعادة تعيينه بعد التعريف
*/

// PI = 3.0;  ❌ خطأ! const لا يتغير

// var - القديمة (تجنبها في المشاريع الجديدة)
var oldWay = 'قديم';
/*
  var كانت تُستخدم قديماً
  لها مشاكل في النطاق (Scope)
  ✅ استخدم let وconst دائماً
*/

🔢 أنواع البيانات البدائية

// 1. String (نص)
let name = 'أحمد';           // بين علامات تنصيص مفردة
let greeting = "مرحباً";    // أو مزدوجة
let message = `مرحباً ${name}!`; // Template Literal - الأقواس الخلفية
/*
  Template Literal مميز لأنه:
  - يقبل متغيرات بين ${}
  - يقبل أسطر متعددة بدون \n
*/

// 2. Number (رقم)
let age = 25;          // عدد صحيح
let price = 99.99;     // عدد عشري
let negative = -10;    // سالب
let big = 1_000_000;   // يمكن كتابة الشرطة السفلية للوضوح

// 3. Boolean (منطقي)
let isLoggedIn = true;   // صح
let isAdmin = false;     // خطأ
// مفيد للشروط والقرارات

// 4. undefined (غير معرّف)
let x;
console.log(x);  // undefined - معرّف لكن لا قيمة

// 5. null (فارغ متعمد)
let user = null;  // فارغ عن قصد - "لا يوجد مستخدم"
/*
  الفرق: undefined = ما اتحدد
          null = حددنا إنه فارغ
*/

🔍 الكشف عن نوع البيانات

// typeof يعيد نوع المتغير كنص
console.log(typeof 'أحمد');   // "string"
console.log(typeof 42);       // "number"
console.log(typeof true);     // "boolean"
console.log(typeof undefined);// "undefined"
console.log(typeof null);     // "object" ← خطأ مشهور في JS!
console.log(typeof []);       // "object"
console.log(typeof {});       // "object"
console.log(typeof function(){});// "function"

🔄 تحويل أنواع البيانات

// نص إلى رقم
let str = '42';
let num = Number(str);      // 42 (رقم)
let num2 = parseInt(str);   // 42 (عدد صحيح)
let num3 = parseFloat('3.14'); // 3.14 (عشري)
let num4 = +str;            // 42 (اختصار)

console.log(Number('hello')); // NaN = Not a Number
// NaN تعني "ليس رقماً"

// رقم إلى نص
let n = 42;
let s1 = String(n);   // "42"
let s2 = n.toString();// "42"
let s3 = `${n}`;      // "42"
let s4 = n + '';      // "42" (ضم مع نص فارغ)

// إلى Boolean
Boolean(0);       // false (الصفر = false)
Boolean('');      // false (نص فارغ = false)
Boolean(null);    // false
Boolean(undefined);// false
Boolean(42);      // true (أي رقم غير صفر)
Boolean('نص');    // true (أي نص غير فارغ)

🎓 ما تعلمناه في هذا الدرس

  • ✅ let للمتغيرات القابلة للتغيير، const للثوابت
  • ✅ أنواع البيانات: String وNumber وBoolean وundefined وnull
  • ✅ Template Literals لدمج النصوص والمتغيرات بـ ${}
  • ✅ typeof لمعرفة نوع المتغير
  • ✅ Number() وString() وBoolean() لتحويل الأنواع
  • ✅ NaN يعني "ليس رقماً" وهو نتيجة تحويل فاشل
